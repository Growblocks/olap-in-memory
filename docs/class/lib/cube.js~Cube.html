    public query(dimensionIds: Array<number>, filter: Object<string, Array<string>>, withTotals: boolean): Object | number source

    Query this cube splitting the result by the provided dimension ids and filters.

    Override:

    AbstractCube#query

    Params:

    NameTypeAttributeDescription
    dimensionIds Array<number>

    Dimension to distribute the result
    filter Object<string, Array<string>>

    Elements that should be included by dimension. Missing dimensions are not filtered.
    withTotals boolean
    • optional
    • default: false

    Include an additional "total" key at every level

    Return:

    Object | number

    An nested object which contain total value in each dimensionElement.

    Example:

    var c = new Cube(...);

c.query();
// 178

c.query(['year']);
// {2013: 25, 2014: 63, 2015: 90}

c.query(['year', 'location']);
// {2013: {shopA: 10, shopB: 20}, 2014: {shopA: 30, shopB: 15}, 2015: {shopA: 43, shopB: 60}}

c.query(['year', 'location'], {year: ['2013', '2014']}, true);
// {2013: {shopA: 10, shopB: 20, _total: 30}, 2014: {shopA: 30, shopB: 15, _total: 45}, _total: 75}
